#include <iostream>
using namespace std;

int main() {
    int a, b, c;

    cout << "Birinci ededi daxil edin: ";
    cin >> a;

    cout << "Ikinci ededi daxil edin: ";
    cin >> b;

    cout << "Emel secin (1-toplama, 2-cixma, 3-vurma, 4-bolme): ";
    cin >> c;

    switch (c) {
        case 1:
            cout << "Netice: " << a + b << endl;
            break;
        case 2:
            cout << "Netice: " << a - b << endl;
            break;
        case 3:
            cout << "Netice: " << a * b << endl;
            break;
        case 4:
            if (b != 0)
                cout << "Netice: " << a / b << endl;
            else
                cout << "Sifira bolmek olmaz!" << endl;
            break;
        default:
            cout << "Yanlis secim etdiniz!" << endl;
    }

    return 0;
}
