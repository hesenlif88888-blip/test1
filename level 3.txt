    cin >> n;

    if (n <= 1)
        cout << "Sade eded deyil." << endl;
    else {
        int i;
        for (i = 2; i < n; i++) {
            if (n % i == 0)
                break;
        }

        if (i == n)
            cout << "Sade ededdir." << endl;
        else
            cout << "Sade eded deyil." << endl;
    }

    return 0;
}

int main() {
    int n;
    cout << "Element sayi: ";
    cin >> n;

    int a = 0, b = 1, c, i = 1;

    cout << "Fibonacci ardicilligi: ";

    do {
        cout << a << " ";
        c = a + b;
        a = b;
        b = c;
        i++;
    } while (i <= n);

    return 0;
}
